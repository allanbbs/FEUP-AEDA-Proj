30 double Servicos::cal_tempo() {
31  double velocity = 60;
32  double multiplier = PI / 180.0;
33  double x1 = departure.get_Coordx() * multiplier;
34  double x2 = arrival.get_Coordx() * multiplier; // get the coordinates of the locals (arrival and departure)
35  double y1 = departure.get_Coordy() * multiplier;
36  double y2 = arrival.get_Coordy() * multiplier;
37  double dy = y2 - y1;
38  double r = 6378.137; //radius of the earth
39 
40  double distance = acos(sin(x1)*sin(x2)+cos(x1)*cos(x2)*cos(dy))*r;
41 
42  // it calculates the distance between the two points
43  return distance / velocity; //calculates the time in hours
44 }
